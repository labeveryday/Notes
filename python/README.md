### Unpacking arguments

```python
numbers = [1, 9]
num_dict = {"x": 5, "y": 9}

def add(x, y):
    return x + y

add(*numbers)

add(**num_dict)


def multiply(*args):
    total = 1
    for arg in args:
        total = total * arg
    return total

def apply(*apply(*args, operator)):
    if operator == "*":
        return multiply(*args)
    elif operator == "+":
        return sum(*args)
    else:
        return "No valid operator provided to apply()"

apply(1, 3, 5, 8, operator="+")
```
